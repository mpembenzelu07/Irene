Developer API

Build on top of our platform with our comprehensive REST API and SDKs 


from pamoja_sdk import PamojaClient

# Initialize client with API key
client = PamojaClient(api_key="pk_live_...")

# Get personalized recommendations
recommendations = client.get_recommendations(
    user_id="user_123",
    location=[-6.7924, 39.2083],  # Dar es Salaam
    budget_range=[5000, 20000]
)

# Create multi-restaurant order
order = client.create_order(
    items=[
        {"restaurant_id": "rest_001", "dish_id": "dish_456"},
        {"restaurant_id": "rest_002", "dish_id": "dish_789"}
    ],
    delivery_location=[-6.7924, 39.2083]
)

# Track order freshness in real-time
tracking = client.track_order(
    order_id=order.id,
    include_freshness=True,
    include_route=True
)

print(f"Order freshness: {tracking.freshness_score}%")
print(f"Estimated delivery: {tracking.eta}")
